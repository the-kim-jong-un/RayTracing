72  Vector3<T> rotateAround(const char &axis, const float &angle, const char &degOrRad='d') {
73  Vector3 tmp;
74  float effAngle;
75  if (degOrRad=='d'){
76  effAngle = angle * (M_PI/180);
77  }
78  switch (axis) {
79  case 'x':
80  tmp.x = x;
81  tmp.y = y * cosf(effAngle) - z * sinf(effAngle);
82  tmp.z = y * sinf(effAngle) + z * cosf(effAngle);
83  return tmp;
84  case 'y':
85  tmp.x = (x * cosf(effAngle)) + (z * sinf(effAngle));
86  tmp.y = y;
87  tmp.z = (-x * sinf(effAngle)) + (z * cosf(effAngle));
88  return tmp;
89  case 'z':
90  tmp.x = (x * cosf(effAngle)) - (y * sinf(effAngle));
91  tmp.y = (x * sinf(effAngle)) + (y * cosf(effAngle));
92  tmp.z = z;
93  return tmp;
94  default:
95  std::cout<<"invalid axis in rotateAround"<<'\n';
96  assert(false);
97  return tmp;
